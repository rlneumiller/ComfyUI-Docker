# neu-comfy-ui-podman
# Build image with ContainerFile:
# podman build -t comfyui-neu:latest .

# To start container normally (access via http)
podman run -it --rm \
  --name comfyui-max \
  --device nvidia.com/gpu=all \
  --security-opt label=disable \
  --userns=keep-id \
  -p 8188:8188 \
  -e COMFYUI_USE_XFORMERS=1 \
  -e TORCH_BACKEND_ENABLE_CUDA_FLASH_ATTENTION=1 \
  -e PYTORCH_CUDA_ALLOC_CONF=max_split_size_mb:256 \
  -v ~/comfyui/models:/workspace/ComfyUI/models:Z \
  -v ~/comfyui/output:/workspace/ComfyUI/output:Z \
  -v ~/comfyui/custom_nodes:/workspace/ComfyUI/custom_nodes:Z \
  -v ~/comfyui/settings:/workspace/ComfyUI/user:Z \
  -v ~/comfyui/temp:/workspace/ComfyUI/temp:Z \
  localhost/comfyui-neu:latest

In the event of a new nvidia driver update, may need to manually update nvidia container toolkit "sudo nvidia-ctk cdi generate --output=/etc/cdi/nvidia.yaml"
